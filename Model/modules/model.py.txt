%matplotlib inline

class Model():
  def __init__(self):
    load_model()
  
  def load_model(self):
    self.__model = hub.load("https://tfhub.dev/tensorflow/ssd_mobilenet_v2/2")

  def pred(self, input_image):
    return self.__model(input_image.get_data())
